- name: "Task 1 — Configure L2 access VLANs on NX-OS CE switches"
  hosts: nxos
  gather_facts: false
  vars:
    vlan_config:
      n9k-ce01:
        id: ___          # TODO: VLAN ID for west-side clients (see Table 1)
        name: "___"      # TODO: Name this VLAN (convention: CLIENT-VLAN-<id>)
      n9k-ce02:
        id: ___          # TODO: VLAN ID for east-side clients (see Table 1)
        name: "___"      # TODO: Name this VLAN (convention: CLIENT-VLAN-<id>)

  tasks:
    - name: "Step 1 — Create VLAN on each CE switch"
      cisco.nxos.nxos_vlans:
        config:
          - vlan_id: {{ vlan_config[inventory_hostname].id }}
            name: {{ vlan_config[inventory_hostname].name }}
            state: active
        state: merged

    - name: "Step 2 — Set Eth1/3 and Eth1/4 to Layer 2 (switchport) mode"
      cisco.nxos.nxos_interfaces:
        config:
          - name: Ethernet1/3
            mode: layer2
          - name: Ethernet1/4
            mode: layer2
        state: merged

    - name: "Step 3 — Assign access VLAN to Eth1/3 and Eth1/4"
      cisco.nxos.nxos_l2_interfaces:
        config:
          - name: Ethernet1/3
            access:
              vlan: {{ vlan_config[inventory_hostname].id }}
          - name: Ethernet1/4
            access:
              vlan: {{ vlan_config[inventory_hostname].id }}
        state: merged

    - name: "Step 4 — Save running configuration"
      cisco.nxos.nxos_config:
        save_when: modified
