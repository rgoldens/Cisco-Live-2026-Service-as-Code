# ── Linux client 1 ───────────────────────────────────────────────────────────
resource "docker_container" "linux1" {
  name  = "linux-terraform1"
  image = var.linux_image

  privileged = true
  restart    = "no"

  networks_advanced {
    name         = docker_network.terraform_net.name
    ipv4_address = var.linux1_ip
  }

  # Write authorized_keys and start sshd
  command = [
    "sh", "-c", join(" && ", [
      "echo 'https://dl-cdn.alpinelinux.org/alpine/v3.12/main' > /etc/apk/repositories",
      "echo 'https://dl-cdn.alpinelinux.org/alpine/v3.12/community' >> /etc/apk/repositories",
      "apk add --no-cache openssh",
      "ssh-keygen -A",
      "mkdir -p /root/.ssh",
      "printf '%s\\n' '${replace(trimspace(var.authorized_keys), \"\\n\", \"' '\")}' > /root/.ssh/authorized_keys",
      "chmod 700 /root/.ssh",
      "chmod 600 /root/.ssh/authorized_keys",
      "echo 'PermitRootLogin yes' > /etc/ssh/sshd_config",
      "echo 'PubkeyAuthentication yes' >> /etc/ssh/sshd_config",
      "echo 'AuthorizedKeysFile .ssh/authorized_keys' >> /etc/ssh/sshd_config",
      "echo 'root:root' | chpasswd",
      "/usr/sbin/sshd",
      "tail -f /dev/null",
    ])
  ]
}
