# ── Terraform Demo — Root ────────────────────────────────────────────────────
# Manages the full terraform-lab stack:
#   1. docker-infra module  — Docker network, volume, and 3 containers
#   2. iosxe-config module  — IOS XE hostname + Loopback0 via RESTCONF
#
# Usage:
#   terraform init
#   terraform apply          # spins up everything (~7 min for CSR to boot)
#   terraform destroy        # tears down everything including containers
#

terraform {
  required_providers {
    docker = {
      source  = "kreuzwerker/docker"
      version = "~> 3.9.0"
    }
    iosxe = {
      source  = "CiscoDevNet/iosxe"
      version = "~> 0.16.0"
    }
  }
}

provider "docker" {
  host = "unix:///var/run/docker.sock"
}

provider "iosxe" {
  host      = module.docker_infra.csr_ip
  username  = var.csr_username
  password  = var.csr_password
  insecure  = true
  protocol  = "restconf"
}

# ── Module: Docker Infrastructure ────────────────────────────────────────────
module "docker_infra" {
  source = "./modules/docker-infra"

  network_name      = var.network_name
  network_subnet    = var.network_subnet
  csr_ip            = var.csr_ip
  csr_username      = var.csr_username
  csr_password      = var.csr_password
  linux1_ip         = var.linux1_ip
  linux2_ip         = var.linux2_ip
  authorized_keys   = var.authorized_keys
  csr_image         = var.csr_image
  linux_image       = var.linux_image
}

# ── Module: IOS XE Configuration ─────────────────────────────────────────────
module "iosxe_config" {
  source = "./modules/iosxe-config"

  hostname       = var.csr_hostname
  loopback_ip    = var.loopback_ip
  loopback_mask  = var.loopback_mask

  # Wait for docker-infra (and its internal CSR readiness wait) before
  # attempting any RESTCONF calls.
  depends_on = [module.docker_infra]
}
