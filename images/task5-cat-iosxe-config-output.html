# ── Module: iosxe-config ─────────────────────────────────────────────────────
# Configures IOS XE device settings via RESTCONF.
# Provider is configured in the root module.
# ─────────────────────────────────────────────────────────────────────────────

terraform {
  required_providers {
    iosxe = {
      source  = "CiscoDevNet/iosxe"
      version = "~> 0.16.0"
    }
  }
}

# Set the device hostname
resource "iosxe_system" "this" {
  hostname = var.hostname
}

# Create Loopback0
resource "iosxe_interface_loopback" "lo0" {
  name              = 0
  description       = "Managed by Terraform"
  ipv4_address      = var.loopback_ip
  ipv4_address_mask = var.loopback_mask
}
